def generate_number_matrix(R, C):
    # 计数器
    count = 1

    # 最大宽度，用于对齐数字
    max_width = len(str(R * C))
    
    # 循循环遍历每行
    for i in range(R):
        # 初始化行字符串
        row_str = ""

        # 循环遍历每列
        for j in range(C):
            # 格式化当前数字并添加到行字符串
            row_str += f"{count:{max_width}} "
            count += 1

        # 打印当前行字符串
        print(R,C)
        print(row_str.strip())

if __name__ == "__main__":
    # 获取用户输入的整数R和C
    R, C = map(int, input().split())

    # 调用生成矩阵的函数
    generate_number_matrix(R, C)
